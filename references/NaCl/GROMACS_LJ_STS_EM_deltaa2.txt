4) deltaA2.sh
-------------
#!/bin/bash

##################################
#Script to compute Delta A_2 term#
##################################

# Variable definitions
### N = number of k' values
### nmol = number of atoms
### cexp = exp(c), where c=3.5 is a constant needed to integrate numerically the Eq.(39)
### of [C.Vega et al., J. Phys. Cond. Matt., 20, 153101 (2008)] (see also Ref.
### [Undertanding Molecular Simulation of D. Frenkel and B. Smit, second Edition, page 260]) 
N=15
T=240.
nmol=256
cexp=33.11545

# Remove old files
rm pesos.dat calcula.dat
rm -r ./director
rm energies.dat integral.dat
mkdir ./director

(for ((i=1; i<$N+1; i++)); do

# Read the value of k' and the Gaussian quadrature weight
 read num landa w
 #echo $landa $w
 echo $w >> pesos.dat

 mkdir ./director/$landa

# Change the value of the spring constant on the topol.top file
 head -25 topol.orig > topol.top
 echo "   1    1    $landa  $landa  $landa ">> topol.top
 tail -9 topol.orig >> topol.top

# Execute GROMACS
 /usr/local/bin/grompp_d -maxwarn 1 &> grompp.log
 /usr/local/bin/mdrun_d -c confout.g96

# Compute the position restraint energy using the g_energy GROMACS's tool 
 ./extrae_a2.sh > kk

 tail -3 kk > kk2
 tail -1 kk2 > save

 read numkk kk3 ener kk4 kk5 kk6 kk7 < save
# Save the energy of the configuration in the energies.dat file
 echo $landa $ener >> energies.dat 

# Remove useless files
 rm energy.xvg

 mv confout.g96 topol.top topol.tpr ./director/$landa
 mv traj.* ./director/$landa
 mv ener.edr md.log mdout.mdp ./director/$landa
 rm state.cpt grompp.log kk kk2 save
 rm state*

done) < landas.dat

#Compute the mean square displacements from the position restraint energies.
awk -v temp=$T mol=$nmol'{print ($1/2.)*(10/(8.314*temp)),$2/($1/2.)*(100/mol)}' energies.dat > integral.dat

paste integral.dat pesos.dat > calcula.dat

# Compute the Delta A_2 term.
# The \sum[msd*(k'*exp(c))*w], where msd is the mean squere displament and w the Gaussian quadrature weight of k'
awk -v const=$cexp '{s+=($2*($1+const))*$3} END {print s*(-1)}' calcula.dat > final_value_DA2.txt
read Da2 < final_value_DA2.txt

echo "Delta A_2 / NkT =" $Da2

rm pesos.dat calcula.dat

# "The devils are in the details
# and looks like the devils are here to stay"
##########################################################################################
